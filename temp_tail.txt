            AddMessageToChat(ok ? $"Switched to {_currentMode} (backend: {best})" : $"Failed to select backend for {_currentMode}", "system");
            UpdateModelStatus();
        }

        private async void ManualModelSelector_SelectionChanged(object sender, SelectionChangedEventArgs e)
        {
            if (_llmService == null || ManualModelSelector.SelectedItem == null) return;
            var model = ((ComboBoxItem)ManualModelSelector.SelectedItem).Content.ToString();
            var ok = await _llmService.SwitchModelAsync(model);
            AddMessageToChat(ok ? $"Manual: switched to {model}" : $"Manual switch failed: {model}", "system");
        }

        private string ChooseBestModelForMode(string mode)
        {
            var available = _llmService.AvailableModels?.ToList() ?? new List<string>();
            string[] priority = new[]
            {
                "Anthropic Claude 3.5 Sonnet",
                "OpenAI GPT-4",
                "Azure OpenAI GPT-4",
                "OpenAI GPT-3.5-Turbo",
                "Advanced AI Assistant",
                "Local Ollama Model",
                "Google Gemini",
                "System Expert Mode"
            };
            foreach (var p in priority) if (available.Contains(p)) return p;
            return _llmService.CurrentModel ?? "Advanced AI Assistant";
        }
